function adicionar(){

let descricao = document.getElementById("descricao").value

function mostrar(){

let lista = document.getElementById("lista")
lista.innerHTML=""

dados.forEach(d=>{

lista.innerHTML+=`
<div class="item">
${d.descricao}
<span>R$ ${d.valor}</span>
</div>
`

})
